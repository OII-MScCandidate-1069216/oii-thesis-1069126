xgb_spec <- boost_tree(

desp_wf <- workflow() %>%
  add_formula(LOCALIZADO_VIVO_MUERTO ~ .)

desp_wf

xgb_rs <- desp_wf %>%
  add_model(xgb_spec) %>%
  fit_resamples(
    resamples = desp_validation,
    control = control_resamples(save_pred = TRUE),
    metrics = metric_set(roc_auc, accuracy, f_meas, bal_accuracy)
  )
